Hello world

  • A first example using data from maptools (ISO2-codes)
library(maptools)
data("wrld_simpl")
ISO2codes <- wrld_simpl@data$ISO2
countries <- c("FR","DE","AT","CH")
ind <- match(countries,ISO2codes)
my_map <- wrld_simpl[ind,]

Plot my map

plot(my_map)
